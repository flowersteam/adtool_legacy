178    def forward(self, inputs):
179        """
180        :param inputs: tensor of size (batch_size, n_inputs)
181        Note: output of a node as follows: activation(bias+(responseaggregation(inputs)))
182        returns: (batch_size, n_outputs)
183        """
184        batch_size = len(inputs)
185        if inputs.device != self.device:
186            inputs = inputs.to(self.device)
187        after_pass_hidden_activs = torch.zeros_like(self.hidden_activs.detach()).to(
188            self.device
189        )
190        after_pass_output_activs = torch.zeros_like(self.output_activs.detach()).to(
191            self.device
192        )
193
194        # mask connection in case sparse 0 values have been changed during gradient descent
195        self.input_to_hidden.data *= self.input_to_hidden_mask
196        self.hidden_to_hidden.data *= self.hidden_to_hidden_mask
197        self.output_to_hidden.data *= self.output_to_hidden_mask
198        self.input_to_output.data *= self.input_to_output_mask
199        self.hidden_to_output.data *= self.hidden_to_output_mask
200        self.output_to_output.data *= self.output_to_output_mask
201
202        # Step 1: multiply incoming inputs with connection weights
203        after_pass_hidden_activs = (
204            inputs.mm(self.input_to_hidden)
205            + self.hidden_activs.mm(self.hidden_to_hidden)
206            + self.output_activs.mm(self.output_to_hidden)
207        )  # batch_size, n_hidden
208
209        after_pass_output_activs = (
210            inputs.mm(self.input_to_output)
211            + self.hidden_activs.mm(self.hidden_to_output)
212            + self.output_activs.mm(self.output_to_output)
213        )  # batch_size, n_outputs
214
215        # Step 2: x = agg(x)
216        # here sum aggregation already happened in Step 1 within torch.mm
217
218        # Step 3: x = bias + reponse * x
219        after_pass_hidden_activs = (
220            self.hidden_biases + self.hidden_responses * after_pass_hidden_activs
221        )
222        after_pass_output_activs = (
223            self.output_biases + self.output_responses * after_pass_output_activs
224        )
225
226        # Step 3: x = act(x)
227        for hidden_neuron_idx in range(self.n_hidden):
228            after_pass_hidden_activs[:, hidden_neuron_idx] = self.hidden_activations[
229                hidden_neuron_idx
230            ](after_pass_hidden_activs[:, hidden_neuron_idx].clone())
231        for output_neuron_idx in range(self.n_outputs):
232            after_pass_output_activs[:, output_neuron_idx] = self.output_activations[
233                output_neuron_idx
234            ](after_pass_output_activs[:, output_neuron_idx].clone())
235
236        self.hidden_activs = self.hidden_mask * after_pass_hidden_activs
237        self.output_activs = self.output_mask * after_pass_output_activs
238
239        return self.output_activs

241    def activate(self, inputs, n_passes=1):
242        """
243        :param inputs: tensor of size (..., n_inputs) => only the last dim matter, then CPPN is applied on all coordinates
244        :param n_passes: number of passes in the RNN TODO: currently global passes should it be internal passes as in pytorch-neat by uber research ?
245        returns: (..., n_outputs)
246        """
247        input_size = inputs.shape[:-1]
248        batch_size = torch.prod(torch.tensor(input_size))
249        assert inputs.shape[-1] == self.n_inputs
250
251        if isinstance(inputs, torch.Tensor):
252            forward_inputs = inputs.view(-1, self.n_inputs)
253        elif use_Minkowski_inputs:
254            forward_inputs = inputs.features
255
256        self.hidden_activs = torch.zeros((batch_size, self.n_hidden)).to(self.device)
257        self.output_activs = torch.zeros((batch_size, self.n_outputs)).to(self.device)
258
259        for _ in range(n_passes):
260            outputs = self.forward(forward_inputs)
261
262        if isinstance(inputs, torch.Tensor):
263            outputs = outputs.view(input_size + (self.n_outputs,))
264        elif use_Minkowski_inputs:
265            outputs = ME.SparseTensor(
266                outputs,
267                coordinate_map_key=inputs.coordinate_map_key,
268                coordinate_manager=inputs.coordinate_manager,
269            )
270
271        return outputs

547    def forward(self, inputs):
548        """
549        :param inputs: tensor of size (batch_size, n_inputs)
550
551        Note: output of a node as follows: activation(bias+(responseaggregation(inputs)))
552
553        returns: (batch_size, n_outputs)
554        """
555        batch_size = len(inputs)
556        if inputs.device != self.device:
557            inputs = inputs.to(self.device)
558        after_pass_node_activs = torch.zeros_like(self.node_activs).to(self.device)
559
560        for neuron_key in self.output_neuron_ids + self.hidden_neuron_ids:

180        """
181        :param inputs: tensor of size (batch_size, n_inputs)
182        Note: output of a node as follows: activation(bias+(responseaggregation(inputs)))
183        returns: (batch_size, n_outputs)
184        """
185        batch_size = len(inputs)
186        if inputs.device != self.device:
187            inputs = inputs.to(self.device)
188        after_pass_hidden_activs = torch.zeros_like(self.hidden_activs.detach()).to(
189            self.device
190        )
191        after_pass_output_activs = torch.zeros_like(self.output_activs.detach()).to(
192            self.device
193        )
194
195        # mask connection in case sparse 0 values have been changed during gradient descent
196        self.input_to_hidden.data *= self.input_to_hidden_mask
197        self.hidden_to_hidden.data *= self.hidden_to_hidden_mask
198        self.output_to_hidden.data *= self.output_to_hidden_mask
199        self.input_to_output.data *= self.input_to_output_mask
200        self.hidden_to_output.data *= self.hidden_to_output_mask
201        self.output_to_output.data *= self.output_to_output_mask
202
203        # Step 1: multiply incoming inputs with connection weights
204        after_pass_hidden_activs = (
205            inputs.mm(self.input_to_hidden)
206            + self.hidden_activs.mm(self.hidden_to_hidden)
207            + self.output_activs.mm(self.output_to_hidden)
208        )  # batch_size, n_hidden
209
210        after_pass_output_activs = (
211            inputs.mm(self.input_to_output)
212            + self.hidden_activs.mm(self.hidden_to_output)
213            + self.output_activs.mm(self.output_to_output)
214        )  # batch_size, n_outputs
215
216        # Step 2: x = agg(x)
217        # here sum aggregation already happened in Step 1 within torch.mm
218
219        # Step 3: x = bias + reponse * x
220        after_pass_hidden_activs = (
221            self.hidden_biases + self.hidden_responses * after_pass_hidden_activs
222        )
223        after_pass_output_activs = (
224            self.output_biases + self.output_responses * after_pass_output_activs
225        )
226

228        for hidden_neuron_idx in range(self.n_hidden):

230                hidden_neuron_idx

239
